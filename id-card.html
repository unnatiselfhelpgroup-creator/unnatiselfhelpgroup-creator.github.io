async function previewPhoto(input) { const file = input.files[0]; if (!file) return; const photoPlaceholder = document.getElementById("photoPlaceholder"); photoPlaceholder.innerHTML = "⏳ प्रोसेसिंग..."; try { // 1. फाइल को सीधे बिटमैप में बदलें (यह सबसे सुरक्षित तरीका है) const bitmap = await createImageBitmap(file); // 2. कैनवास पर ड्रा करें const canvas = document.createElement("canvas"); const MAX_WIDTH = 500; const scale = MAX_WIDTH / bitmap.width; canvas.width = MAX_WIDTH; canvas.height = bitmap.height * scale; const ctx = canvas.getContext("2d"); ctx.drawImage(bitmap, 0, 0, canvas.width, canvas.height); // 3. डेटा सुरक्षित करें window.photoDataURL = canvas.toDataURL("image/jpeg", 0.7); document.getElementById("photoPreview").src = window.photoDataURL; document.getElementById("photoPreview").style.display = "block"; photoPlaceholder.style.display = "none"; // मेमोरी खाली करें bitmap.close(); } catch (err) { console.error("Error:", err); alert("फ़ाइल लोड नहीं हो सकी। कृपया किसी अन्य गैलरी या इमेज का उपयोग करें।"); photoPlaceholder.innerHTML = "📷 फोटो अपलोड करने के लिए यहाँ क्लिक करें"; } }
