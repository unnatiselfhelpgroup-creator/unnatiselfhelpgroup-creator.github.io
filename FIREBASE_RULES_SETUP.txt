━━━ FIRESTORE RULES ━━━
Firebase Console → Firestore → Rules में यह paste करें:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null &&
             request.auth.token.email == "unnatiselfhelpgroup@gmail.com";
    }

    match /StudentRegistrations/{docId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if isAdmin();
    }

    match /Appointments/{docId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if isAdmin();
    }

    match /Volunteers/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /DailyReports/{docId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if isAdmin();
    }

    match /Certificates/{docId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if isAdmin();
    }

    match /IDcard/{docId} {
      allow create: if true;
      allow read: if true;
      allow update, delete: if isAdmin();
    }

    // ✅ Sequential Counter (appointment letter numbers)
    match /Counters/{docId} {
      allow read: if true;
      allow write: if true;  // counter publicly writable (create only needed)
    }
  }
}

━━━ FIREBASE STORAGE RULES ━━━
Firebase Console → Storage → Rules में यह paste करें:

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // ✅ Payment Proofs - User upload कर सकता है, Admin पढ़ सकता है
    match /payment-proofs/{allPaths=**} {
      allow read: if request.auth != null;   // सिर्फ Admin देख सकता है
      allow write: if request.resource.size < 5 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }

    // RTI Documents - Public read
    match /rti-docs/{fileName} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
